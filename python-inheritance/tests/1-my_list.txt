>>> MyList = __import__('1-my_list').MyList

# Basic usage
>>> my_list = MyList([1, 4, 2, 3, 5])
>>> print(my_list)
[1, 4, 2, 3, 5]
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]
>>> print(my_list)
[1, 4, 2, 3, 5]

# Check append()
>>> my_list = MyList([1, 2])
>>> my_list.append(3)
>>> print(my_list)
[1, 2, 3]

# Type checks (inheritance validation)
>>> isinstance(my_list, MyList)
True
>>> isinstance(my_list, list)
True

# Unsorted input remains unchanged after print_sorted
>>> my_list = MyList([3, 1, 2])
>>> my_list.print_sorted()
[1, 2, 3]
>>> print(my_list)
[3, 1, 2]

# Empty list
>>> my_list = MyList()
>>> my_list.print_sorted()
[]
>>> print(my_list)
[]

# Multiple calls should not modify original list
>>> my_list = MyList([5, 4, 3])
>>> my_list.print_sorted()
[3, 4, 5]
>>> my_list.print_sorted()
[3, 4, 5]
>>> print(my_list)
[5, 4, 3]

# Negative numbers (edge case required by checker)
>>> my_list = MyList([3, -1, 2, -5])
>>> print(my_list)
[3, -1, 2, -5]
>>> my_list.print_sorted()
[-5, -1, 2, 3]
>>> print(my_list)
[3, -1, 2, -5]
